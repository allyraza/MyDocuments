Method name convention

? method name end with "?" will return a boolean value

! method name end with "!" will modify the receiver

= method name end with "=" can appear on left side of assignment

Method arguments

def my_method(arg1="default",arg2="default")
  puts "#{arg1}, #{arg2}"
end

my_method("see") => see, default

using default arguments

 

def my_method(*arg)
  arg.each {|item| puts item}
end

my_method("1","2","3") => 1,2,3

using dynamic arguments

 

class MyClass
  def initialize(&block) # add & for blcok argument
    @block = block
  end

  def my_method(arg)
    @block.call(arg) # execuate the block
  end
end

i_variable = MyClass.new() { |item| puts item } # pass block as argument
i_variable.my_method("see") => see

using block as argument
