Create sub class

class Parent
  def initialize
    @greeting = "Hello "
  end
end

class Child < Parent
  def initialize(name)
    @name = name
    super() # call constructor in parent class to create greeting
  end

  def say_hello
    puts @greeting + @name
  end
end

my_variable = Child.new("harry")
my_variable.say_hello => Hello harry

Access control

  • Public method: can be access by everyone. Methods are public by default
  • Protected methods: can by access by other instance of same class or its sub class.
  • Private methods: self

class MyClass
  protected # subsequent methods will be 'protected'
    def method1
    #...
    end

  private # subsequent methods will be 'private'
    def method3
    #...
    end

  public # subsequent methods will be 'public'
    def method5
    #...
    end
end

require 'csv' # include Ruby library
require_relative 'book_in_stock' #include another class

class CsvReader
  def initialize
    @books_in_stock = [] # create empty array
  end

  def read_in_csv_data(csv_file_name)
    CSV.foreach(csv_file_name, headers: true) do |row| # loop data from CSV file
      @books_in_stock << BookInStock.new(row["ISBN"], row["Amount"]) # create BookInStock instance and push to empty array
    end
  end
end
